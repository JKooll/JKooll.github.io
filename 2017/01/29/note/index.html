#include <stdio.h>
#define SIZE 1000002

long int table[SIZE];

long int fn(long int n)
{
    if(n < SIZE && table[n] > 0) {
        return table[n];
    }

    if(n & 1) {//odd
        if(n < SIZE) {
            table[n] = 2 + fn((3 * n + 1) >> 1);

            return table[n];
        }

        return 2 + fn((3 * n + 1) >> 1);
    } else {
        if(n < SIZE) {
            table[n] = 1 + fn(n >> 1);

            return table[n];
        }

        return 1 + fn(n >> 1);
    }
}

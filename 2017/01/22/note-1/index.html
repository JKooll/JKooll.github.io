拼接两个json对象

//拼接两个json对象
function merge_jsonobject(jsonobj1, jsonobj2)
{
    let result_json = {};

    for(let attr in jsonobj1) {
        result_json[attr] = jsonobj1[attr];
    }

    for(let attr in jsonobj2) {
        result_json[attr] = jsonobj2[attr];
    }

    return result_json;
}
